📊 Connexion Google Sheets

Le site lit ses données directement depuis un Google Sheets public. Pour mettre à jour le site : modifie le tableur, puis clique Recharger sur le site. C'est tout !

1
Dans ton Google Sheets, crée ces 6 onglets (noms exacts) :
config  actualites  grille  resultats  classement  pilotes
2
Dans le Sheet : Extensions → Apps Script. Supprime tout le code existant, colle le script ci-dessous et clique 💾 Enregistrer. 
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheets = ['config','actualites','grille','resultats','classement','pilotes'];
  var out = {};
  sheets.forEach(function(name) {
    var s = ss.getSheetByName(name);
    if (!s) { out[name] = []; return; }
    var vals = s.getDataRange().getValues();
    if (vals.length < 2) { out[name] = []; return; }
    var headers = vals[0];
    out[name] = vals.slice(1).filter(function(r) {
      return r.some(function(c) { return c !== ''; });
    }).map(function(r) {
      var o = {};
      headers.forEach(function(h, i) { if (h) o[String(h)] = String(r[i] || ''); });
      return o;
    });
  });
  var json = JSON.stringify(out);
  var cb = e && e.parameter && e.parameter.callback;
  if (cb) {
    return ContentService.createTextOutput(cb + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var cols = {
      config:     ['cle','valeur'],
      actualites: ['titre','categorie','date','emoji','description','une'],
      grille:     ['position','numero','pilote','equipe','constructeur','temps','ecart'],
      resultats:  ['position','numero','pilote','equipe','constructeur','temps','points'],
      classement: ['position','numero','pilote','equipe','points'],
      pilotes:    ['numero','nom','equipe','nationalite','drapeau']
    };
    Object.keys(cols).forEach(function(name) {
      if (!data[name]) return;
      var sheet = ss.getSheetByName(name);
      if (!sheet) return;
      var headers = cols[name];
      sheet.clearContents();
      sheet.appendRow(headers);
      data[name].forEach(function(row) {
        sheet.appendRow(headers.map(function(h) { return row[h] || ''; }));
      });
    });
    return ContentService.createTextOutput('{"ok":true}')
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput('{"ok":false,"error":"' + err.message + '"}')
      .setMimeType(ContentService.MimeType.JSON);
  }
}
